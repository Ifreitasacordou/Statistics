def produtoCartesiano(c1, c2):
    r = []
    x = []
    y = []
    for i in range(len(c1)):
        for j in range(len(c2)):
            r.append([c1[i], c2[j]])    
            
    for i in range(len(r)):
        x.append([r[i][0]])
    for i in range(len(r)):
        y.append([r[i][1]])
        
    plt.plot(x, y, 'bo')
    plt.show()
    return print("O produto cartesiano √©: ", r)
